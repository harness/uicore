# MultiLevelSelect --- docs/pages/multiselect.mdx

## Import

```js
import { MultiLevelSelect } from '@wings-software/uikit'
```

## Example

```js

interface Option{
    label :string
    value: string
    options?: Option[]
}

const options : Option[] = [
  { value: 'application', label: 'Application',
  options: [
    { value: 'name', label: 'Name' },
    { value: 'id', label: 'ID' },
    { value: 'tag', label: 'Tag' },
    { value: 'otherAttribute', label: 'Other Attribute' },
  ]
},
{ value: 'service', label: 'Service',
options: [
  { value: 'name', label: 'Name' },
  { value: 'id', label: 'ID' },
  { value: 'tag', label: 'Tag' },
  { value: 'otherAttribute', label: 'Other Attribute' },
]
},
  { value: 'artifactServer', label: 'Atrifact Server'}
]

 <MultiLevelSelect options={options} placeholder={'Select...'} onChange={() => {}}/>

```

## Reference

This component allows selection from multiple levels.

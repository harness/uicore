# Card Icon --- docs/pages/card-icon.mdx

Card Icon component reuses Card Component and lets you to write body for icon [CD Next Gen](https://harness.atlassian.net/wiki/spaces/UI/pages/668271019/Card)

## Import

```js
import { Card, CardBody } from '@wings-software/uicore'
```

## Types

```js
interface CardProps {
  elevation: number (0|1|2|3|4);
  interactive: boolean (false);
  selected: boolean (false);
  onClick?: (e: React.MouseEvent<HTMLDivElement>) => void;
}

// Card Body CardProps
interface CardIconProps extends HTMLDivProps {
  icon: IconName
  iconSize?: number
  iconProps?: Partial<IconProps>
}

// Example
<Card interactive="true" selected>
    <CardBody.Icon icon="service-kubernetes" iconSize={25}>
        <Text font={{size: 'small', align: 'center', color: 'var(--grey-900)'}}>Kubernetes</Text>
    </CardBody.Icon>
</Card>
```

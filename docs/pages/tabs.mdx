# Tabs --- docs/pages/tabs.mdx

## Import

```js
import { Tabs } from '@harness/uicore'
import { Tab } from '@harness/uicore'
```

## Single Tab

```js live=true
<Layout.Horizontal spacing="small">
  <Tabs id="tabsId1">
    <Tab id="tabId1" title="Tab Item" panel="Tab Content" />
  </Tabs>
</Layout.Horizontal>
```

## Multiple Tabs

```js live=true
<Layout.Horizontal spacing="small">
  <Tabs id="tabsId1">
    <Tab id="tabId1" title="Tab Item 1" panel="Tab 1 Content" />
    <Tab id="tabId2" title="Tab Item 2" panel="Tab 2 Content" />
    <Tab id="tabId3" title="Tab Item 3" panel="Tab 3 Content" />
    <Tab id="tabId4" title="Tab Item 4" panel="Tab 4 Content" />
    <Tab id="tabId5" title="Tab Item 5" panel="Tab 5 Content" />
  </Tabs>
</Layout.Horizontal>
```
